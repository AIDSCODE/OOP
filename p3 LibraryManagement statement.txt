import java.util.Scanner;

class Book
{
int bookId;
String bookName;
String author;
boolean issued;

static int totalBooks = 0;

Book(int id, String name, String auth)
{
bookId = id;
bookName = name;
author = auth;
issued = false;
totalBooks++;
}

void displayBook()
{
System.out.println("\nBook ID     : " + bookId);
System.out.println("Book Name   : " + bookName);
System.out.println("Author Name : " + author);

if(issued)
{
System.out.println("Status      : Issued");
}
else
{
System.out.println("Status      : Available");
}
}

void issueBook()
{
if(!issued)
{
issued = true;
System.out.println("\nBook Issued Successfully");
}
else
{
System.out.println("\nBook Already Issued");
}
}

void returnBook()
{
if(issued)
{
issued = false;
System.out.println("\nBook Returned Successfully");
}
else
{
System.out.println("\nBook Was Not Issued");
}
}

static void showTotalBooks()
{
System.out.println("\nTotal Books : " + totalBooks);
}
}

class LibraryManagement
{
public static void main(String[] args)
{
Scanner sc = new Scanner(System.in);

Book b1 = new Book(101, "Java Programming", "James Gosling");

int choice;

do
{
System.out.println("\n===== LIBRARY MENU =====");
System.out.println("1. View Book Details");
System.out.println("2. Issue Book");
System.out.println("3. Return Book");
System.out.println("4. Show Total Books");
System.out.println("5. Exit");

System.out.print("Enter Your Choice: ");
choice = sc.nextInt();

switch(choice)
{
case 1:
b1.displayBook();
break;

case 2:
b1.issueBook();
break;

case 3:
b1.returnBook();
break;

case 4:
Book.showTotalBooks();
break;

case 5:
System.out.println("\nThanks For Using Library Management System!");
break;

default:
System.out.println("\nInvalid Choice");
}
}
while(choice != 5);

sc.close();
}
}
